CREATE DATABASE ConstructionDB;

USE ConstructionDB;

GO
 

CREATE TABLE [EmployeeDetail](
    [EmployeeID] VARCHAR(20) NOT NULL,
    [Department] NVARCHAR(20) NOT NULL,
	[ImmediateSupervisor] VARCHAR(20),
    [EmployeePhoto] VARBINARY(MAX) NOT NULL,
	[EmployeeName] NVARCHAR(20) NOT NULL,
    [PhoneNumber] VARCHAR(20) NOT NULL UNIQUE,
    [Mail] VARCHAR(50) NOT NULL UNIQUE,
    [Password] VARCHAR(100) NOT NULL,
    [Address] NVARCHAR(100) NOT NULL,
    [Gender] NCHAR(1) NOT NULL,
    [BirthDate] DATETIME NOT NULL,
	[EmergencyContact] NVARCHAR(20) NOT NULL,
	[EmergencyRelationship] NVARCHAR(10) NOT NULL,
	[EmergencyContactPhone] NVARCHAR(20) NOT NULL,
	[AnnualLeave] TINYINT NOT NULL,
    PRIMARY KEY([EmployeeID])
);
GO

CREATE TABLE [Temporarier] (
	[EmployeeID] VARCHAR(20) NOT NULL,
	[EmployeeName] NVARCHAR(20) NOT NULL,
	[PhoneNumber] VARCHAR(20) NOT NULL,
	[CompanyID] VARCHAR(3) NOT NULL,
	PRIMARY KEY([EmployeeID])
);
GO

CREATE TABLE [Company] (
	[CompanyID] VARCHAR(3) NOT NULL,
	[CompanyName] NVARCHAR(50) NOT NULL UNIQUE,
	PRIMARY KEY([CompanyID])
);
GO

CREATE TABLE [Employee] (
	[EmployeeID] VARCHAR(20) NOT NULL,
	[IsActive] BIT NOT NULL,
    [Position] NVARCHAR(20) NOT NULL,
	[PositionRole] NVARCHAR(10) NOT NULL,
	[HireDate] DATETIME NOT NULL,
	[ResignationDate] DATETIME,
	PRIMARY KEY([EmployeeID])
);
GO

ALTER TABLE [Employee]
ADD CONSTRAINT CHK_ResignationDate CHECK (ResignationDate IS NULL OR ResignationDate >= HireDate);


CREATE TABLE [AttendanceRecord] (
	[AttendanceID] VARCHAR(20) NOT NULL,
	[EmployeeID] VARCHAR(20) NOT NULL,
	[WorkDate] DATETIME NOT NULL,
	[CheckInTime] DATETIME NOT NULL,
	[CheckOutTime] DATETIME,
	PRIMARY KEY([AttendanceID])
);
GO

CREATE TABLE [LeaveRecord] (
	[EmployeeID] VARCHAR(20) NOT NULL,
	[StartTime] DATETIME NOT NULL,
	[EndTime] DATETIME NOT NULL,
	[SpentHours] TINYINT NOT NULL,
	[LeaveStatus] BIT NOT NULL,
	PRIMARY KEY([EmployeeID], [StartTime])
);
GO

CREATE TABLE [Accident] (
	[AccidentID] VARCHAR(20) NOT NULL UNIQUE,	
	[AccidentType] NVARCHAR(10) NOT NULL,
	[AccidentTitle] NVARCHAR(10) NOT NULL,
	[Description] NVARCHAR(100) NOT NULL,
	[StartTime] DATETIME NOT NULL,
	[EmployeeID] VARCHAR(20) NOT NULL,
    [UploadTime] DATETIME NOT NULL,
    [IncidentControllerID] VARCHAR(20) NOT NULL,
	[Response] NVARCHAR(100),
	[EndTime] DATETIME,
	[IncidentStatus] BIT NOT NULL,
	PRIMARY KEY([AccidentID])
);
GO

CREATE TABLE [AccidentFile] (
	[FileID] NVARCHAR(20) NOT NULL,
	[AccidentID] VARCHAR(20) NOT NULL,
	[FileImage] VARBINARY(MAX) NOT NULL,
	[ResponseType] BIT NOT NULL,
	PRIMARY KEY([FileID])
);
GO


CREATE TABLE [Blueprint] (
	[BlueprintID] NVARCHAR(20) NOT NULL UNIQUE,
    [PrintCategoryID] VARCHAR(3) NOT NULL,
	[BlueprintName] NVARCHAR(50) NOT NULL,
	[BlueprintVersion] DECIMAL(10,1) NOT NULL,
	[BlueprintFile] VARBINARY(MAX) NOT NULL,
	[Description] NVARCHAR(100),
	[UploadTime] DATETIME NOT NULL,
    [EmployeeID] VARCHAR(20) NOT NULL,
    [PrintStatus] BIT,
	PRIMARY KEY([BlueprintID])
);
GO

--PrintCategory => PrintCategoryName
CREATE TABLE [PrintCategory] (
	[PrintCategoryID] VARCHAR(3) NOT NULL,
	[PrintCategory] NVARCHAR(10) NOT NULL UNIQUE,
	PRIMARY KEY([PrintCategoryID])
);
GO


-- 事故檔案與事故紀錄的關聯
ALTER TABLE [AccidentFile]
ADD CONSTRAINT FK_AccidentFile_Accident
FOREIGN KEY([AccidentID]) REFERENCES [Accident]([AccidentID])
ON UPDATE NO ACTION ON DELETE CASCADE;
GO

ADD CONSTRAINT FK_AttendanceRecord_Employee
FOREIGN KEY([EmployeeID]) REFERENCES [Employee]([EmployeeID])
ON UPDATE NO ACTION ON DELETE NO ACTION;
GO

FOREIGN KEY([EmployeeID]) REFERENCES [Employee]([EmployeeID])
ON UPDATE NO ACTION ON DELETE NO ACTION;
GO

-- 員工與請假紀錄的關聯

ADD CONSTRAINT FK_LeaveRecord_Employee
FOREIGN KEY([EmployeeID]) REFERENCES [Employee]([EmployeeID])
ON UPDATE NO ACTION ON DELETE NO ACTION;
GO

-- 員工與出勤紀錄的關聯
ALTER TABLE [Accident]
ADD CONSTRAINT FK_Accident_Employee
FOREIGN KEY([EmployeeID]) REFERENCES [Employee]([EmployeeID])
ON UPDATE NO ACTION ON DELETE NO ACTION;
GO


-- 臨時員工與員工表的關聯

FOREIGN KEY([EmployeeID]) REFERENCES [Employee]([EmployeeID])
ON UPDATE NO ACTION ON DELETE NO ACTION;
GO
